    # replace_images.py として保存
import re
import json

def replace_single_base64(content, base64_image, image_number, log_file):
    """
    1つのBase64画像を置換し、ログを記録
    """
    # 置換前の状態を記録
    before_lines = content.count('\n')

    # 置換実行
    new_content = content.replace(base64_image, f"![][{image_number}]", 1)

    # 検証
    if content == new_content:
        print(f"❌ エラー: 置換が実行されませんでした")
        return None

    after_lines = new_content.count('\n')
    if before_lines != after_lines:
        print(f"❌ エラー: 行数が変更されました")
        return None

    # ログ記録
    log_entry = {
        "timestamp": datetime.now().isoformat(),
        "image_number": image_number,
        "status": "success"
    }
    log_file.write(json.dumps(log_entry) + "\n")

    print(f"✅ {image_number} の置換完了")
    return new_content

    5.1 自動検証¶

    # final_verify.py として保存
def verify_final_file(file_path):
    """
    最終ファイルの完全検証
    """
    errors = []
    warnings = []

    with open(file_path, 'r', encoding='utf-8') as f:
        content = f.read()
        lines = content.split('\n')

    # 1. Base64画像の残存チェック
    if 'data:image' in content:
        errors.append("Base64画像が残っています")

    # 2. 画像参照の検証
    img_refs = re.findall(r'!\[\]\[(image\d+)\]', content)
    for img_ref in img_refs:
        if not re.search(rf'^\[{img_ref}\]:', content, re.MULTILINE):
            errors.append(f"{img_ref} の定義がありません")

    # 3. 画像定義の検証
    definitions = re.findall(r'^\[(image\d+)\]:', content, re.MULTILINE)
    for definition in definitions:
        if img_refs.count(definition[1:-2]) == 0:
            warnings.append(f"{definition[1:-2]} は使用されていません")

    # 4. キャプションの検証
    captions = [line for line in lines if line.strip().startswith('図')]
    if len(captions) != 19:
        warnings.append(f"キャプション数が期待値と異なります: {len(captions)}")

    return errors, warnings

# 実行
errors, warnings = verify_final_file('final.md')
if errors:
    print("❌ エラーが見つかりました:")
    for error in errors:
        print(f"  - {error}")
else:
    print("✅ エラーなし")

if warnings:
    print("⚠️ 警告:")
    for warning in warnings:
        print(f"  - {warning}")
