    🚀 実行コマンド¶

    レビュー完了後、以下のコマンドで修正を実行：

    # 1. 作業ディレクトリへ移動
cd C:\Users\xprin\github\flying-robot-contest-rules-public\docs

# 2. バックアップ作成（PowerShell）
$timestamp = Get-Date -Format "yyyyMMdd_HHmmss"
Copy-Item "21FlyRobo_GeneralAutoUniqueMulti_Regulations_text_EDIT.md" `
          "temp\backups\21FlyRobo_GeneralAutoUniqueMulti_Regulations_text_EDIT_$timestamp.md"
Copy-Item "index.md" "temp\backups\index_$timestamp.md"
Copy-Item "..\mkdocs.yml" "temp\backups\mkdocs_$timestamp.yml"

# 3. 修正スクリプト実行
python temp\correct\execute_full_revision.py

# 4. MkDocsでプレビュー確認
mkdocs serve
