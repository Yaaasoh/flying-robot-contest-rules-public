def test_replacements():
    """置換結果をテスト"""
    with open('21FlyRobo_GeneralAutoUniqueMulti_Regulations_text_WORK.md', 'r', encoding='utf-8') as f:
        content = f.read()

    errors = []
    # 古い形式が残っていないか確認
    if '![][image' in content:
        errors.append("❌ 古い画像参照が残っています")

    # 新しい形式が正しく挿入されているか確認（統一形式）
    for img_id, path in IMAGE_MAPPING.items():
        if path:  # 削除対象でない場合
            expected = f"![]({path})"
            if expected not in content:
                errors.append(f"❌ {img_id}: 期待される形式が見つかりません")

    return errors

    # remove_image_definitions.py
def remove_definitions(file_path):
    """ファイル末尾の[imageN]: path形式を削除"""
    with open(file_path, 'r', encoding='utf-8') as f:
        lines = f.readlines()

    # 画像定義行を除外
    new_lines = []
    for line in lines:
        if not re.match(r'^\[image\d+\]:', line):
            new_lines.append(line)

    with open(file_path, 'w', encoding='utf-8') as f:
        f.writelines(new_lines)
